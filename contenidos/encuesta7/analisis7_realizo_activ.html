def resp_num_p6(var_col_name_1, 
                df_egresados_sorted):
        names                = []
        asistencia_count     = []
        presentaciones_count = []
        publicaciones_count  = []
        
        var_col_name_1 = '¿Tuvo la oportunidad de realizar alguna de las actividades que a continuación se indican? '
        title_col_name_1 = '¿Tuvo la oportunidad de realizar alguna de las <br> actividades que a continuación se indican? '
        names.append(title_col_name_1)
        cols_to_use  = [var_col_name_1]
        for col in cols_to_use:
            cnt_srs = df_egresados_sorted[col].value_counts()
            asistencia_count.append(cnt_srs['Asistencia'])
            presentaciones_count.append(cnt_srs['Presentaciones'])
            publicaciones_count.append(cnt_srs['Publicaciones'])
    
        trace01 = go.Bar(
            y=names,
            x=asistencia_count,
            orientation = 'h',
            name = "Asistencia"
        )    
        trace11 = go.Bar(
            y=names,
            x=presentaciones_count,
            orientation = 'h',
            name = "Presentaciones"
        )
        trace21 = go.Bar(
            y=names,
            x=publicaciones_count,
            orientation = 'h',
            name = "Publicaciones"
        )
        
        layout = go.Layout(
        title='' ,
        barmode = 'stack',
        width   = 750,
        height  = 250
        )
        
        data = [trace01, trace11, trace21]
        fig = go.Figure(data=data, layout=layout)

        fig.update_layout(
        font_family="Rockwell",
        font_color="black",
        font_size=12,
        title_font_family="Rockwell", title_font_size= 14,
        title_font_color="black",
        legend_title_font_color="green",
        )
        py.iplot(fig, filename="WorkToolsFrequency")
